import type { Request, Response } from "express";
import Stripe from "stripe";

const stripe = new Stripe(process.env.STRIPE_SECRET_KEY!, {
  apiVersion: "2024-06-20",
});

// Map your plan selections -> lookup_key created by your script
const LOOKUP_KEY: Record<string, string> = {
  "basic:monthly": "fnb_basic_monthly",
  "basic:quarterly": "fnb_basic_quarterly",
  "basic:annual": "fnb_basic_annual",
  "pro:monthly": "fnb_pro_monthly",
  "pro:quarterly": "fnb_pro_quarterly",
  "pro:annual": "fnb_pro_annual",
};

function requireEnv(name: string) {
  const v = process.env[name];
  if (!v) throw new Error(`Missing env var: ${name}`);
  return v;
}

/**
 * POST /api/billing/checkout
 * Body: { tier: "basic"|"pro", term: "monthly"|"quarterly"|"annual", email?: string, userId?: string }
 *
 * Returns: { url }
 */
export async function createCheckoutSession(req: Request, res: Response) {
  try {
    requireEnv("STRIPE_SECRET_KEY");
    const baseUrl = requireEnv("APP_BASE_URL");

    const { tier, term, email, userId } = req.body ?? {};
    if (!tier || !term) return res.status(400).json({ message: "tier and term required" });

    const key = `${tier}:${term}`;
    const lookupKey = LOOKUP_KEY[key];
    if (!lookupKey) return res.status(400).json({ message: "Invalid tier/term" });

    // Fetch price by lookup_key (works in test & live)
    const prices = await stripe.prices.search({
      query: `lookup_key:'${lookupKey}' AND active:'true'`,
      limit: 1,
    });
    const price = prices.data[0];
    if (!price) return res.status(404).json({ message: `No active price for ${lookupKey}` });

    const session = await stripe.checkout.sessions.create({
      mode: "subscription",
      line_items: [{ price: price.id, quantity: 1 }],
      success_url: `${baseUrl}/signup/success?session_id={CHECKOUT_SESSION_ID}`,
      cancel_url: `${baseUrl}/signup/cancel`,
      customer_email: email || undefined,

      // Optional: attach your internal user id so webhook can activate the right account
      client_reference_id: userId || undefined,

      // Optional: useful metadata
      metadata: {
        tier,
        term,
        lookup_key: lookupKey,
      },
    });

    return res.json({ url: session.url });
  } catch (err: any) {
    console.error("createCheckoutSession error:", err);
    return res.status(500).json({ message: "Failed to create checkout session" });
  }
}
